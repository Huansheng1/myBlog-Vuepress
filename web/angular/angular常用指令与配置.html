ng g m views/pages/demoPage --routing
1

# 创建组件时不生成测试文件

ng generate component my-component --no-spec
1

# 常用配置

# 修改全局组件开头

系统默认的是app-，我们通过修改tslint.json文件可将默认开头改为我们自己想要的：

// 像这类我们创建默认开头就变成了kt
{
  "extends": "../tslint.json",
  "rules": {
    "directive-selector": [
      true,
      "attribute",
      "kt",
      "camelCase"
    ],
    "component-selector": [
      true,
      "element",
      "kt",
      "kebab-case"
    ]
  }
}
